1. Створіть і використайте функцію, що будує такий же атрибут, як і в попередньому завданні: функція має приймати на вхід значення року, а повертати різницю в роках між поточною датою та датою, створеною з атрибута року (1996 рік → ‘1996-01-01’).

DELIMITER //

CREATE FUNCTION YearsDifference(input_year TEXT)
RETURNS TEXT
DETERMINISTIC 
NO SQL
BEGIN
    DECLARE result TEXT;
    SET result = TIMESTAMPDIFF(YEAR,MAKEDATE(input_year,1), curdate());
    RETURN result;
END //

DELIMITER ;

SELECT YearsDifference(`Year`) as `Years difference` from infections;
